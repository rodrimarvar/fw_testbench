50//! @private @memberof QActive
51bool QActive_post_(QActive * const me,
52 QEvt const * const e,
53 uint_fast16_t const margin,
54 void const * const sender)
55{
56#ifdef Q_UTEST // test?
57#if (Q_UTEST != 0) // testing QP-stub?
58 if (me->super.temp.fun == Q_STATE_CAST(0)) { // QActiveDummy?
59 return QActiveDummy_fakePost_(me, e, margin, sender);
60 }
61#endif // (Q_UTEST != 0)
62#endif // def Q_UTEST
63
64 QF_CRIT_STAT
65 QF_CRIT_ENTRY();
66
67 Q_REQUIRE_INCRIT(200, e != (QEvt *)0);
68
69 QEQueueCtr const nFree = me->eQueue.nFree; // get volatile into temporary
70
71 // required margin available?
72 bool status = false; // assume that event cannot be posted
73 if (margin == QF_NO_MARGIN) {
74 if (nFree > 0U) { // free entries available in the queue?
75 status = true; // can post
76 }
77 else { // no free entries available
78 // The queue overflows, but QF_NO_MARGIN indicates that
79 // the "event delivery guarantee" is required.
80 Q_ERROR_INCRIT(210); // must be able to post the event
81 }
82 }
83 else if (nFree > (QEQueueCtr)margin) { // enough free entries?
84 status = true; // can post
85 }
86 else {
87 // empty
88 }
89
90#if (QF_MAX_EPOOL > 0U)
91 if (e->poolNum_ != 0U) { // is it a mutable event?
92 Q_ASSERT_INCRIT(205, e->refCtr_ < (2U * QF_MAX_ACTIVE));
93 QEvt_refCtr_inc_(e); // increment the reference counter
94 }
95#endif // (QF_MAX_EPOOL > 0U)
96
97 if (status) { // can post the event?
98 QActive_postFIFO_(me, e, sender);
99 QF_CRIT_EXIT();
100 }
101 else { // event cannot be posted
102 QS_BEGIN_PRE(QS_QF_ACTIVE_POST_ATTEMPT, me->prio)
103 QS_TIME_PRE(); // timestamp
104 QS_OBJ_PRE(sender); // the sender object
105 QS_SIG_PRE(e->sig); // the signal of the event
106 QS_OBJ_PRE(me); // this active object (recipient)
107 QS_2U8_PRE(e->poolNum_, e->refCtr_);
108 QS_EQC_PRE(nFree); // # free entries
109 QS_EQC_PRE(margin); // margin requested
110 QS_END_PRE()
111
112#ifdef Q_UTEST
113 if (QS_LOC_CHECK_(me->prio)) {
114 QF_CRIT_EXIT();
115 QS_onTestPost(sender, me, e, status); // QUTEst callback
116 QF_CRIT_ENTRY();
117 }
118#endif // def Q_USTEST
119
120 QF_CRIT_EXIT();
121
122#if (QF_MAX_EPOOL > 0U)
123 QF_gc(e); // recycle the event to avoid a leak
124#endif // (QF_MAX_EPOOL > 0U)
125 }
126
127 return status;
128}

131//! @private @memberof QActive
132void QActive_postLIFO_(QActive * const me,
133 QEvt const * const e)
134{
135#ifdef Q_UTEST // test?
136#if (Q_UTEST != 0) // testing QP-stub?
137 if (me->super.temp.fun == Q_STATE_CAST(0)) { // QActiveDummy?
138 QActiveDummy_fakePostLIFO_(me, e);
139 return;
140 }
141#endif // (Q_UTEST != 0)
142#endif // def Q_UTEST
143
144 QF_CRIT_STAT
145 QF_CRIT_ENTRY();
146
147 // the posted event must be be valid (which includes not NULL)
148 Q_REQUIRE_INCRIT(300, e != (QEvt *)0);
149
150 QEQueueCtr tmp = me->eQueue.nFree; // get volatile into temporary
151
152 // The queue must NOT overflow for the LIFO posting policy.
153 Q_REQUIRE_INCRIT(310, tmp != 0U);
154
155 if (e->poolNum_ != 0U) { // is it a mutable event?
156 Q_ASSERT_INCRIT(305, e->refCtr_ < (2U * QF_MAX_ACTIVE));
157 QEvt_refCtr_inc_(e); // increment the reference counter
158 }
159
160 --tmp; // one free entry just used up
161
162 me->eQueue.nFree = tmp; // update the original
163 if (me->eQueue.nMin > tmp) {
164 me->eQueue.nMin = tmp; // update minimum so far
165 }
166
167 QS_BEGIN_PRE(QS_QF_ACTIVE_POST_LIFO, me->prio)
168 QS_TIME_PRE(); // timestamp
169 QS_SIG_PRE(e->sig); // the signal of this event
170 QS_OBJ_PRE(me); // this active object
171 QS_2U8_PRE(e->poolNum_, e->refCtr_);
172 QS_EQC_PRE(tmp); // # free entries
173 QS_EQC_PRE(me->eQueue.nMin); // min # free entries
174 QS_END_PRE()
175
176#ifdef Q_UTEST
177 // callback to examine the posted event under the same conditions
178 // as producing the #QS_QF_ACTIVE_POST trace record, which are:
179 // the local filter for this AO ('me->prio') is set
180 if (QS_LOC_CHECK_(me->prio)) {
181 QF_CRIT_EXIT();
182
183 QS_onTestPost((QActive *)0, me, e, true);
184
185 QF_CRIT_ENTRY();
186 }
187#endif // def Q_UTEST
188
189 QEvt const * const frontEvt = me->eQueue.frontEvt;
190 me->eQueue.frontEvt = e; // deliver the event directly to the front
191
192 if (frontEvt != (QEvt *)0) { // was the queue NOT empty?
193 tmp = me->eQueue.tail; // get volatile into temporary;
194 ++tmp;
195 if (tmp == me->eQueue.end) { // need to wrap the tail?
196 tmp = 0U; // wrap around
197 }
198 me->eQueue.tail = tmp;
199 me->eQueue.ring[tmp] = frontEvt;
200 }
201 else { // queue was empty
202 QACTIVE_EQUEUE_SIGNAL_(me); // signal the event queue
203 }
204
205 QF_CRIT_EXIT();
206}

209//! @private @memberof QActive
210QEvt const * QActive_get_(QActive * const me) {
211 QF_CRIT_STAT
212 QF_CRIT_ENTRY();
213
214 // wait for event to arrive directly (depends on QP port)
215 // NOTE: might use assertion-IDs 400-409
216 QACTIVE_EQUEUE_WAIT_(me);
217
218 // always remove event from the front
219 QEvt const * const e = me->eQueue.frontEvt;
220 QEQueueCtr tmp = me->eQueue.nFree; // get volatile into temporary
221
222 Q_REQUIRE_INCRIT(410, e != (QEvt *)0); // queue must NOT be empty
223
224 ++tmp; // one more free event in the queue
225
226 me->eQueue.nFree = tmp; // update the # free
227
228 if (tmp <= me->eQueue.end) { // any events in the ring buffer?
229
230 QS_BEGIN_PRE(QS_QF_ACTIVE_GET, me->prio)
231 QS_TIME_PRE(); // timestamp
232 QS_SIG_PRE(e->sig); // the signal of this event
233 QS_OBJ_PRE(me); // this active object
234 QS_2U8_PRE(e->poolNum_, e->refCtr_);
235 QS_EQC_PRE(tmp); // # free entries
236 QS_END_PRE()
237
238 // remove event from the tail
239 tmp = me->eQueue.tail; // get volatile into temporary
240 QEvt const * const frontEvt = me->eQueue.ring[tmp];
241
242 Q_ASSERT_INCRIT(430, frontEvt != (QEvt *)0);
243 me->eQueue.frontEvt = frontEvt; // update the original
244
245 if (tmp == 0U) { // need to wrap the tail?
246 tmp = me->eQueue.end;
247 }
248 --tmp; // advance the tail (counter-clockwise)
249
250 me->eQueue.tail = tmp; // update the original
251 }
252 else {
253 me->eQueue.frontEvt = (QEvt *)0; // queue becomes empty
254
255 // all entries in the queue must be free (+1 for fronEvt)
256 Q_ASSERT_INCRIT(440, tmp == (me->eQueue.end + 1U));
257
258 QS_BEGIN_PRE(QS_QF_ACTIVE_GET_LAST, me->prio)
259 QS_TIME_PRE(); // timestamp
260 QS_SIG_PRE(e->sig); // the signal of this event
261 QS_OBJ_PRE(me); // this active object
262 QS_2U8_PRE(e->poolNum_, e->refCtr_);
263 QS_END_PRE()
264 }
265
266 QF_CRIT_EXIT();
267
268 return e;
269}

272//! @private @memberof QActive
273static void QActive_postFIFO_(QActive * const me,
274 QEvt const * const e,
275 void const * const sender)
276{
277 // NOTE: this helper function is called *inside* critical section
278#ifndef Q_SPY
279 Q_UNUSED_PAR(sender);
280#endif
281
282 QEQueueCtr tmp = me->eQueue.nFree; // get volatile into temporary
283 --tmp; // one free entry just used up
284
285 me->eQueue.nFree = tmp; // update the original
286 if (me->eQueue.nMin > tmp) {
287 me->eQueue.nMin = tmp; // update minimum so far
288 }
289
290 QS_BEGIN_PRE(QS_QF_ACTIVE_POST, me->prio)
291 QS_TIME_PRE(); // timestamp
292 QS_OBJ_PRE(sender); // the sender object
293 QS_SIG_PRE(e->sig); // the signal of the event
294 QS_OBJ_PRE(me); // this active object (recipient)
295 QS_2U8_PRE(e->poolNum_, e->refCtr_);
296 QS_EQC_PRE(tmp); // # free entries
297 QS_EQC_PRE(me->eQueue.nMin); // min # free entries
298 QS_END_PRE()
299
300#ifdef Q_UTEST
301 if (QS_LOC_CHECK_(me->prio)) {
302 QF_CRIT_EXIT();
303 QS_onTestPost(sender, me, e, true); // QUTEst callback
304 QF_CRIT_ENTRY();
305 }
306#endif // def Q_UTEST
307
308 if (me->eQueue.frontEvt == (QEvt *)0) { // is the queue empty?
309 me->eQueue.frontEvt = e; // deliver event directly
310
311#ifdef QXK_H_
312 if (me->super.state.act == Q_ACTION_CAST(0)) { // eXtended?
313 QXTHREAD_EQUEUE_SIGNAL_(me); // signal eXtended Thread
314 }
315 else {
316 QACTIVE_EQUEUE_SIGNAL_(me); // signal the Active Object
317 }
318#else
319 QACTIVE_EQUEUE_SIGNAL_(me); // signal the Active Object
320#endif // def QXK_H_
321 }
322 else { // queue was not empty, insert event into the ring-buffer
323 tmp = me->eQueue.head; // get volatile into temporary
324 me->eQueue.ring[tmp] = e; // insert e into buffer
325
326 if (tmp == 0U) { // need to wrap the head?
327 tmp = me->eQueue.end;
328 }
329 --tmp; // advance the head (counter-clockwise)
330
331 me->eQueue.head = tmp; // update the original
332 }
333}

392 QAsm * const me,
393 QEvt const * const e,
394 uint_fast8_t const qsId)
395{
396 Q_UNUSED_PAR(e);
397 Q_UNUSED_PAR(qsId);
398
399 QF_CRIT_STAT
400 QF_CRIT_ENTRY();
401
402 // get volatile into temporaries
403 QEQueueCtr nTicks = QACTIVE_CAST_(me)->eQueue.tail;
404 QEQueueCtr const tickRate = QACTIVE_CAST_(me)->eQueue.head;
405

409
410 QF_CRIT_EXIT();

418//! @private @memberof QTicker
419void QTicker_trig_(
420 QActive * const me,
421 void const * const sender)
422{
423#ifndef Q_SPY
424 Q_UNUSED_PAR(sender);
425#endif
426
427 static QEvt const tickEvt = QEVT_INITIALIZER(0);
428
429 QF_CRIT_STAT
430 QF_CRIT_ENTRY();
431
432 QEQueueCtr nTicks = me->eQueue.tail; // get volatile into temporary
433
434 if (me->eQueue.frontEvt == (QEvt *)0) { // no tick events?
435 Q_REQUIRE_INCRIT(900,
436 (me->eQueue.nFree == 1U)
437 && (nTicks == 0U));
438
439 me->eQueue.frontEvt = &tickEvt; // deliver event directly
440 me->eQueue.nFree = 0U;
441
442 QACTIVE_EQUEUE_SIGNAL_(me); // signal the event queue
443 }
444 else {
445 Q_REQUIRE_INCRIT(910, (nTicks > 0U) && (nTicks < 0xFFU));
446 Q_REQUIRE_INCRIT(920, me->eQueue.nFree == 0U);
447 }
448
449 ++nTicks; // account for one more tick event
450
451 me->eQueue.tail = nTicks; // update the original
452
453 QS_BEGIN_PRE(QS_QF_ACTIVE_POST, me->prio)
454 QS_TIME_PRE(); // timestamp
455 QS_OBJ_PRE(sender); // the sender object
456 QS_SIG_PRE(0U); // the signal of the event
457 QS_OBJ_PRE(me); // this active object
458 QS_2U8_PRE(0U, 0U); // poolNum & refCtr
459 QS_EQC_PRE(0U); // # free entries
460 QS_EQC_PRE(0U); // min # free entries
461 QS_END_PRE()
462
463 QF_CRIT_EXIT();
464}

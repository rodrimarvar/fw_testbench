52//! @static @public @memberof QV
53void QV_schedDisable(uint_fast8_t const ceiling) {
54 QF_CRIT_STAT
55 QF_CRIT_ENTRY();
56
57 if (ceiling > QV_priv_.schedCeil) { // raising the scheduler ceiling?
58
59 QS_BEGIN_PRE(QS_SCHED_LOCK, 0U)
60 QS_TIME_PRE(); // timestamp
61 // the previous sched ceiling & new sched ceiling
62 QS_2U8_PRE((uint8_t)QV_priv_.schedCeil,
63 (uint8_t)ceiling);
64 QS_END_PRE()
65
66 QV_priv_.schedCeil = ceiling;
67 }
68 QF_CRIT_EXIT();
69}

72//! @static @public @memberof QV
73void QV_schedEnable(void) {
74 QF_CRIT_STAT
75 QF_CRIT_ENTRY();
76
77 if (QV_priv_.schedCeil != 0U) { // actually enabling the scheduler?
78
79 QS_BEGIN_PRE(QS_SCHED_UNLOCK, 0U)
80 QS_TIME_PRE(); // timestamp
81 // current sched ceiling (old), previous sched ceiling (new)
82 QS_2U8_PRE((uint8_t)QV_priv_.schedCeil, 0U);
83 QS_END_PRE()
84
85 QV_priv_.schedCeil = 0U;
86 }
87 QF_CRIT_EXIT();
88}

112//! @static @public @memberof QF
113int_t QF_run(void) {
114#ifdef Q_SPY
115 // produce the QS_QF_RUN trace record
116 QF_INT_DISABLE();
117 QS_beginRec_((uint_fast8_t)QS_QF_RUN);
118 QS_endRec_();
119 QF_INT_ENABLE();
120#endif // Q_SPY
121
122 QF_onStartup(); // application-specific startup callback
123
124 QF_INT_DISABLE();
125
126#ifdef QV_START
127 QV_START(); // port-specific startup of the QV kernel
128#endif
129
130#if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
131 uint_fast8_t pprev = 0U; // previously used prio.
132
133#ifdef QF_ON_CONTEXT_SW
134 // officially switch to the idle cotext
135 QF_onContextSw((QActive *)0, (QActive *)0);
136#endif // def QF_ON_CONTEXT_SW
137
138#endif // (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
139
140 for (;;) { // QV event loop...
141 // find the maximum prio. AO ready to run
142 uint_fast8_t const p = (QPSet_notEmpty(&QV_priv_.readySet)
143 ? QPSet_findMax(&QV_priv_.readySet)
144 : 0U);
145
146 if (p > QV_priv_.schedCeil) { // is it above the sched ceiling?
147 QActive * const a = QActive_registry_[p];
148
149#if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
150 QS_BEGIN_PRE(QS_SCHED_NEXT, p)
151 QS_TIME_PRE(); // timestamp
152 QS_2U8_PRE((uint8_t)p,
153 (uint8_t)pprev);
154 QS_END_PRE()
155
156#ifdef QF_ON_CONTEXT_SW
157 QF_onContextSw(((pprev != 0U)
158 ? QActive_registry_[pprev]
159 : (QActive *)0), a);
160#endif // QF_ON_CONTEXT_SW
161
162 pprev = p; // update previous prio.
163#endif // (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
164
165 QF_INT_ENABLE();
166
167 QEvt const * const e = QActive_get_(a);
168
169 // dispatch event (virtual call)
170 (*a->super.vptr->dispatch)(&a->super, e, p);
171#if (QF_MAX_EPOOL > 0U)
172 QF_gc(e);
173#endif
174 QF_INT_DISABLE();
175
176 if (a->eQueue.frontEvt == (QEvt *)0) { // empty queue?
177 QPSet_remove(&QV_priv_.readySet, p);
178 }
179 }
180 else { // no AO ready to run --> idle
181#if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
182 if (pprev != 0U) {
183 QS_BEGIN_PRE(QS_SCHED_IDLE, pprev)
184 QS_TIME_PRE(); // timestamp
185 QS_U8_PRE((uint8_t)pprev); // previous prio
186 QS_END_PRE()
187
188#ifdef QF_ON_CONTEXT_SW
189 QF_onContextSw(QActive_registry_[pprev], (QActive *)0);
190#endif // QF_ON_CONTEXT_SW
191
192 pprev = 0U; // update previous prio.
193 }
194#endif // (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
195
196 // QV_onIdle() must be called with interrupts DISABLED because
197 // the determination of the idle condition can change at any time
198 // by an interrupt posting events to a queue.
199 //
200 // NOTE: QV_onIdle() MUST enable interrupts internally, ideally
201 // atomically with putting the CPU into a power-saving mode.
202 QV_onIdle();
203
204 QF_INT_DISABLE(); // disable interrupts before looping back
205 }
206 }
207#ifdef __GNUC__ // GNU compiler?
208 return 0;
209#endif
210}

213//! @public @memberof QActive
214void QActive_start(QActive * const me,
215 QPrioSpec const prioSpec,
216 QEvtPtr * const qSto,
217 uint_fast16_t const qLen,
218 void * const stkSto,
219 uint_fast16_t const stkSize,
220 void const * const par)
221{
222 Q_UNUSED_PAR(stkSto); // not needed in QV
223 Q_UNUSED_PAR(stkSize); // not needed in QV
224
225 QF_CRIT_STAT
226 QF_CRIT_ENTRY();
227 Q_REQUIRE_INCRIT(300, (me->super.vptr != (struct QAsmVtable *)0)
228 && (stkSto == (void *)0));
229 QF_CRIT_EXIT();
230
231 me->prio = (uint8_t)(prioSpec & 0xFFU); // QF-prio. of the AO
232 me->pthre = 0U; // not used
233 QActive_register_(me); // make QF aware of this active object
234
235 QEQueue_init(&me->eQueue, qSto, qLen); // init the built-in queue
236
237 // top-most initial tran. (virtual call)
238 (*me->super.vptr->init)(&me->super, par, me->prio);
239 QS_FLUSH(); // flush the trace buffer to the host
240}

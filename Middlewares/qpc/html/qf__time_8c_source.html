72//! @public @memberof QTimeEvt
73void QTimeEvt_armX(QTimeEvt * const me,
74 uint32_t const nTicks,
75 uint32_t const interval)
76{
77 QF_CRIT_STAT
78 QF_CRIT_ENTRY();
79
80 // dynamic range checks
81#if (QF_TIMEEVT_CTR_SIZE == 1U)
82 Q_REQUIRE_INCRIT(400, (nTicks < 0xFFU) && (interval < 0xFFU));
83#elif (QF_TIMEEVT_CTR_SIZE == 2U)
84 Q_REQUIRE_INCRIT(400, (nTicks < 0xFFFFU) && (interval < 0xFFFFU));
85#endif
86
87 QTimeEvtCtr const ctr = me->ctr;
88 uint8_t const tickRate = me->tickRate;
89#ifdef Q_SPY
90 uint_fast8_t const qsId = ((QActive *)(me->act))->prio;
91#endif // def Q_SPY
92
93 Q_REQUIRE_INCRIT(410,
94 (nTicks != 0U)
95 && (ctr == 0U)
96 && (me->act != (void *)0)
97 && (tickRate < (uint_fast8_t)QF_MAX_TICK_RATE));
98
99#ifdef Q_UNSAFE
100 Q_UNUSED_PAR(ctr);
101#endif // ndef Q_UNSAFE
102
103 me->ctr = (QTimeEvtCtr)nTicks;
104 me->interval = (QTimeEvtCtr)interval;
105
106 // is the time event unlinked?
107 // NOTE: For the duration of a single clock tick of the specified tick
108 // rate a time event can be disarmed and yet still linked into the list
109 // because un-linking is performed exclusively in QTimeEvt_tick_().
110 if ((me->flags & QTE_FLAG_IS_LINKED) == 0U) {
111 me->flags |= QTE_FLAG_IS_LINKED; // mark as linked
112
113 // The time event is initially inserted into the separate
114 // "freshly armed" list based on timeEvtHead_[tickRate].act.
115 // Only later, inside QTimeEvt_tick_(), the "freshly armed"
116 // list is appended to the main list of armed time events based on
117 // timeEvtHead_[tickRate].next. Again, this is to keep any
118 // changes to the main list exclusively inside QTimeEvt_tick_().
119 me->next = (QTimeEvt *)QTimeEvt_timeEvtHead_[tickRate].act;
120 QTimeEvt_timeEvtHead_[tickRate].act = me;
121 }
122
123 QS_BEGIN_PRE(QS_QF_TIMEEVT_ARM, qsId)
124 QS_TIME_PRE(); // timestamp
125 QS_OBJ_PRE(me); // this time event object
126 QS_OBJ_PRE(me->act); // the active object
127 QS_TEC_PRE(nTicks); // the # ticks
128 QS_TEC_PRE(interval); // the interval
129 QS_U8_PRE(tickRate); // tick rate
130 QS_END_PRE()
131
132 QF_CRIT_EXIT();
133}

136//! @public @memberof QTimeEvt
137bool QTimeEvt_disarm(QTimeEvt * const me) {
138 QF_CRIT_STAT
139 QF_CRIT_ENTRY();
140
141 QTimeEvtCtr const ctr = me->ctr;
142
143#ifdef Q_SPY
144 uint_fast8_t const qsId = QACTIVE_CAST_(me->act)->prio;
145#endif
146
147 // was the time event actually armed?
148 bool wasArmed = false;
149 if (ctr != 0U) {
150 wasArmed = true;
151 me->flags |= QTE_FLAG_WAS_DISARMED;
152 me->ctr = 0U; // schedule removal from the list
153
154 QS_BEGIN_PRE(QS_QF_TIMEEVT_DISARM, qsId)
155 QS_TIME_PRE(); // timestamp
156 QS_OBJ_PRE(me); // this time event object
157 QS_OBJ_PRE(me->act); // the target AO
158 QS_TEC_PRE(ctr); // the # ticks
159 QS_TEC_PRE(me->interval); // the interval
160 QS_U8_PRE(me->tickRate); // tick rate
161 QS_END_PRE()
162 }
163 else { // the time event was already disarmed automatically
164 me->flags &= (uint8_t)(~QTE_FLAG_WAS_DISARMED & 0xFFU);
165
166 QS_BEGIN_PRE(QS_QF_TIMEEVT_DISARM_ATTEMPT, qsId)
167 QS_TIME_PRE(); // timestamp
168 QS_OBJ_PRE(me); // this time event object
169 QS_OBJ_PRE(me->act); // the target AO
170 QS_U8_PRE(me->tickRate); // tick rate
171 QS_END_PRE()
172 }
173
174 QF_CRIT_EXIT();
175
176 return wasArmed;
177}

180//! @public @memberof QTimeEvt
181bool QTimeEvt_rearm(QTimeEvt * const me,
182 uint32_t const nTicks)
183{
184 QF_CRIT_STAT
185 QF_CRIT_ENTRY();
186
187 // dynamic range checks
188#if (QF_TIMEEVT_CTR_SIZE == 1U)
189 Q_REQUIRE_INCRIT(600, nTicks < 0xFFU);
190#elif (QF_TIMEEVT_CTR_SIZE == 2U)
191 Q_REQUIRE_INCRIT(600, nTicks < 0xFFFFU);
192#endif
193
194 uint8_t const tickRate = me->tickRate;
195 QTimeEvtCtr const ctr = me->ctr;
196
197 Q_REQUIRE_INCRIT(610,
198 (nTicks != 0U)
199 && (me->act != (void *)0)
200 && (tickRate < QF_MAX_TICK_RATE));
201
202#ifdef Q_SPY
203 uint_fast8_t const qsId = ((QActive *)(me->act))->prio;
204#endif
205
206 me->ctr = (QTimeEvtCtr)nTicks;
207
208 // was the time evt not running?
209 bool wasArmed = false;
210 if (ctr == 0U) {
211 // NOTE: For the duration of a single clock tick of the specified
212 // tick rate a time event can be disarmed and yet still linked into
213 // the list, because unlinking is performed exclusively in the
214 // QTimeEvt_tick_() function.
215
216 // is the time event unlinked?
217 if ((me->flags & QTE_FLAG_IS_LINKED) == 0U) {
218 me->flags |= QTE_FLAG_IS_LINKED; // mark as linked
219
220 // The time event is initially inserted into the separate
221 // "freshly armed" list based on timeEvtHead_[tickRate].act.
222 // Only later, inside QTimeEvt_tick_(), the "freshly armed"
223 // list is appended to the main list of armed time events based on
224 // timeEvtHead_[tickRate].next. Again, this is to keep any
225 // changes to the main list exclusively inside QTimeEvt_tick()_.
226 me->next = (QTimeEvt *)QTimeEvt_timeEvtHead_[tickRate].act;
227 QTimeEvt_timeEvtHead_[tickRate].act = me;
228 }
229 }
230 else { // the time event was armed
231 wasArmed = true;
232 }
233
234 QS_BEGIN_PRE(QS_QF_TIMEEVT_REARM, qsId)
235 QS_TIME_PRE(); // timestamp
236 QS_OBJ_PRE(me); // this time event object
237 QS_OBJ_PRE(me->act); // the target AO
238 QS_TEC_PRE(nTicks); // the # ticks
239 QS_TEC_PRE(me->interval); // the interval
240 QS_2U8_PRE(tickRate, (wasArmed ? 1U : 0U));
241 QS_END_PRE()
242
243 QF_CRIT_EXIT();
244
245 return wasArmed;
246}

286//! @static @private @memberof QTimeEvt
287void QTimeEvt_tick_(
288 uint_fast8_t const tickRate,
289 void const * const sender)
290{
291#ifndef Q_SPY
292 Q_UNUSED_PAR(sender);
293#endif
294
295 QF_CRIT_STAT
296 QF_CRIT_ENTRY();
297
298 Q_REQUIRE_INCRIT(800, tickRate < Q_DIM(QTimeEvt_timeEvtHead_));
299
300 QTimeEvt *prev = &QTimeEvt_timeEvtHead_[tickRate];
301
302#ifdef Q_SPY
303 QS_BEGIN_PRE(QS_QF_TICK, 0U)
304 ++prev->ctr;
305 QS_TEC_PRE(prev->ctr); // tick ctr
306 QS_U8_PRE(tickRate); // tick rate
307 QS_END_PRE()
308#endif // def Q_SPY
309
310 // scan the linked-list of time events at this rate...
311 while (true) {
312 Q_ASSERT_INCRIT(810, prev != (QTimeEvt *)0); // sanity check
313
314 QTimeEvt *te = prev->next; // advance down the time evt. list
315
316 if (te == (QTimeEvt *)0) { // end of the list?
317 // NO any new time events armed since the last QTimeEvt_tick_()?
318 if (QTimeEvt_timeEvtHead_[tickRate].act == (void *)0) {
319 break; // terminate the while-loop
320 }
321
322 prev->next = (QTimeEvt*)QTimeEvt_timeEvtHead_[tickRate].act;
323 QTimeEvt_timeEvtHead_[tickRate].act = (void *)0;
324
325 te = prev->next; // switch to the new list
326 }
327
328 // the time event 'te' must be valid
329 Q_ASSERT_INCRIT(840, te != (QTimeEvt *)0);
330
331 QTimeEvtCtr ctr = te->ctr; // move volatile into temporary
332
333 if (ctr == 0U) { // time event scheduled for removal?
334 prev->next = te->next;
335
336 // mark time event 'te' as NOT linked
337 te->flags &= (uint8_t)(~QTE_FLAG_IS_LINKED & 0xFFU);
338 // do NOT advance the prev pointer
339 QF_CRIT_EXIT(); // exit crit. section to reduce latency
340 }
341 else if (ctr == 1U) { // is time event about to expire?
342 QActive * const act = (QActive *)te->act;
343 prev = QTimeEvt_expire_(te, prev, act, tickRate);
344
345#ifdef QXK_H_
346 if ((enum_t)te->super.sig < Q_USER_SIG) {
347 QXThread_timeout_(act);
348 QF_CRIT_EXIT();
349 }
350 else {
351 QF_CRIT_EXIT(); // exit crit. section before posting
352
353 // QACTIVE_POST() asserts if the queue overflows
354 QACTIVE_POST(act, &te->super, sender);
355 }
356#else // not QXK
357 QF_CRIT_EXIT(); // exit crit. section before posting
358
359 // QACTIVE_POST() asserts if the queue overflows
360 QACTIVE_POST(act, &te->super, sender);
361#endif
362 }
363 else { // time event keeps timing out
364 --ctr; // decrement the tick counter
365 te->ctr = ctr; // update the original
366
367 prev = te; // advance to this time event
368 QF_CRIT_EXIT(); // exit crit. section to reduce latency
369 }
370 QF_CRIT_ENTRY(); // re-enter crit. section to continue the loop
371 }
372 QF_CRIT_EXIT();
373}

397//! @private @memberof QTimeEvt
398QTimeEvt * QTimeEvt_expire_(QTimeEvt * const me,
399 QTimeEvt * const prev_link,
400 QActive const * const act,
401 uint_fast8_t const tickRate)
402{
403 // NOTE: this helper function is called *inside* critical section
404#ifndef Q_SPY
405 Q_UNUSED_PAR(act);
406 Q_UNUSED_PAR(tickRate);
407#endif
408
409 QTimeEvt *prev = prev_link;
410 if (me->interval != 0U) { // periodic time evt?
411 me->ctr = me->interval; // rearm the time event
412 prev = me; // advance to this time event
413 }
414 else { // one-shot time event: automatically disarm
415 me->ctr = 0U;
416 prev->next = me->next;
417
418 // mark this time event as NOT linked
419 me->flags &= (uint8_t)(~QTE_FLAG_IS_LINKED & 0xFFU);
420 // do NOT advance the prev pointer
421
422 QS_BEGIN_PRE(QS_QF_TIMEEVT_AUTO_DISARM, act->prio)
423 QS_OBJ_PRE(me); // this time event object
424 QS_OBJ_PRE(act); // the target AO
425 QS_U8_PRE(tickRate); // tick rate
426 QS_END_PRE()
427 }
428
429 QS_BEGIN_PRE(QS_QF_TIMEEVT_POST, act->prio)
430 QS_TIME_PRE(); // timestamp
431 QS_OBJ_PRE(me); // the time event object
432 QS_SIG_PRE(me->super.sig);// signal of this time event
433 QS_OBJ_PRE(act); // the target AO
434 QS_U8_PRE(tickRate); // tick rate
435 QS_END_PRE()
436
437 return prev;
438}

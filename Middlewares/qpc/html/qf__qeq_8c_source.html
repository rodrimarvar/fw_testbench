44//! @public @memberof QEQueue
45void QEQueue_init(QEQueue * const me,
46 struct QEvt const * * const qSto,
47 uint_fast16_t const qLen)
48{
49 QF_CRIT_STAT
50 QF_CRIT_ENTRY();
51
52#if (QF_EQUEUE_CTR_SIZE == 1U)
53 Q_REQUIRE_INCRIT(100, qLen < 0xFFU);
54#endif
55
56 me->frontEvt = (QEvt *)0; // no events in the queue
57 me->ring = qSto; // the beginning of the ring buffer
58 me->end = (QEQueueCtr)qLen;
59 if (qLen > 0U) {
60 me->head = 0U;
61 me->tail = 0U;
62 }
63 me->nFree = (QEQueueCtr)(qLen + 1U); // +1 for frontEvt
64 me->nMin = me->nFree;
65
66 QF_CRIT_EXIT();
67}

70//! @public @memberof QEQueue
71bool QEQueue_post(QEQueue * const me,
72 struct QEvt const * const e,
73 uint_fast16_t const margin,
74 uint_fast8_t const qsId)
75{
76#ifndef Q_SPY
77 Q_UNUSED_PAR(qsId);
78#endif
79
80 QF_CRIT_STAT
81 QF_CRIT_ENTRY();
82
83 Q_REQUIRE_INCRIT(200, e != (QEvt *)0);
84
85 QEQueueCtr tmp = me->nFree; // get volatile into temporary
86
87 // can the queue accept the event?
88 bool const status = ((margin == QF_NO_MARGIN) && (tmp > 0U))
89 || (tmp > (QEQueueCtr)margin);
90 if (status) {
91 // is it a mutable event?
92 if (e->poolNum_ != 0U) {
93 Q_ASSERT_INCRIT(205, e->refCtr_ < (2U * QF_MAX_ACTIVE));
94 QEvt_refCtr_inc_(e); // increment the reference counter
95 }
96
97 --tmp; // one free entry just used up
98
99 me->nFree = tmp; // update the original
100 if (me->nMin > tmp) { // is this the new minimum?
101 me->nMin = tmp; // update minimum so far
102 }
103
104#ifdef Q_SPY
105 QS_BEGIN_PRE(QS_QF_EQUEUE_POST, qsId)
106 QS_TIME_PRE(); // timestamp
107 QS_SIG_PRE(e->sig); // the signal of the event
108 QS_OBJ_PRE(me); // this queue object
109 QS_2U8_PRE(e->poolNum_, e->refCtr_);
110 QS_EQC_PRE(tmp); // # free entries
111 QS_EQC_PRE(me->nMin); // min # free entries
112 QS_END_PRE()
113#endif // def Q_SPY
114
115 if (me->frontEvt == (QEvt *)0) { // is the queue empty?
116 me->frontEvt = e; // deliver event directly
117 }
118 else { // queue was not empty, insert event into the ring-buffer
119 tmp = me->head; // get volatile into temporary
120 me->ring[tmp] = e; // insert e into buffer
121
122 if (tmp == 0U) { // need to wrap the head?
123 tmp = me->end;
124 }
125 --tmp; // advance head (counter-clockwise)
126
127 me->head = tmp; // update the original
128 }
129 }
130 else { // event cannot be posted
131 // dropping events must be acceptable
132 Q_ASSERT_INCRIT(230, margin != QF_NO_MARGIN);
133
134#ifdef Q_SPY
135 QS_BEGIN_PRE(QS_QF_EQUEUE_POST_ATTEMPT, qsId)
136 QS_TIME_PRE(); // timestamp
137 QS_SIG_PRE(e->sig); // the signal of this event
138 QS_OBJ_PRE(me); // this queue object
139 QS_2U8_PRE(e->poolNum_, e->refCtr_);
140 QS_EQC_PRE(tmp); // # free entries
141 QS_EQC_PRE(margin); // margin requested
142 QS_END_PRE()
143#endif // def Q_SPY
144 }
145
146 QF_CRIT_EXIT();
147
148 return status;
149}

152//! @public @memberof QEQueue
153void QEQueue_postLIFO(QEQueue * const me,
154 struct QEvt const * const e,
155 uint_fast8_t const qsId)
156{
157#ifndef Q_SPY
158 Q_UNUSED_PAR(qsId);
159#endif
160
161 QF_CRIT_STAT
162 QF_CRIT_ENTRY();
163
164 Q_REQUIRE_INCRIT(300, e != (QEvt *)0);
165
166 QEQueueCtr tmp = me->nFree; // get volatile into temporary
167
168 // must be able to LIFO-post the event
169 Q_REQUIRE_INCRIT(310, tmp != 0U);
170
171 if (e->poolNum_ != 0U) { // is it a mutable event?
172 Q_ASSERT_INCRIT(305, e->refCtr_ < (2U * QF_MAX_ACTIVE));
173 QEvt_refCtr_inc_(e); // increment the reference counter
174 }
175
176 --tmp; // one free entry just used up
177
178 me->nFree = tmp; // update the original
179 if (me->nMin > tmp) { // is this the new minimum?
180 me->nMin = tmp; // update minimum so far
181 }
182
183 QS_BEGIN_PRE(QS_QF_EQUEUE_POST_LIFO, qsId)
184 QS_TIME_PRE(); // timestamp
185 QS_SIG_PRE(e->sig); // the signal of this event
186 QS_OBJ_PRE(me); // this queue object
187 QS_2U8_PRE(e->poolNum_, e->refCtr_);
188 QS_EQC_PRE(tmp); // # free entries
189 QS_EQC_PRE(me->nMin); // min # free entries
190 QS_END_PRE()
191
192 QEvt const * const frontEvt = me->frontEvt;
193 me->frontEvt = e; // deliver the event directly to the front
194
195 if (frontEvt != (QEvt *)0) { // was the queue NOT empty?
196 tmp = me->tail; // get volatile into temporary;
197 ++tmp;
198 if (tmp == me->end) { // need to wrap the tail?
199 tmp = 0U; // wrap around
200 }
201 me->tail = tmp;
202 me->ring[tmp] = frontEvt;
203 }
204
205 QF_CRIT_EXIT();
206}

209//! @public @memberof QEQueue
210struct QEvt const * QEQueue_get(QEQueue * const me,
211 uint_fast8_t const qsId)
212{
213#ifndef Q_SPY
214 Q_UNUSED_PAR(qsId);
215#endif
216
217 QF_CRIT_STAT
218 QF_CRIT_ENTRY();
219
220 QEvt const * const e = me->frontEvt; // always remove evt from the front
221
222 if (e != (QEvt *)0) { // was the queue not empty?
223 QEQueueCtr tmp = me->nFree; // get volatile into temporary
224
225 ++tmp; // one more free event in the queue
226
227 me->nFree = tmp; // update the # free
228
229 // any events in the ring buffer?
230 if (tmp <= me->end) {
231
232 QS_BEGIN_PRE(QS_QF_EQUEUE_GET, qsId)
233 QS_TIME_PRE(); // timestamp
234 QS_SIG_PRE(e->sig); // the signal of this event
235 QS_OBJ_PRE(me); // this queue object
236 QS_2U8_PRE(e->poolNum_, e->refCtr_);
237 QS_EQC_PRE(tmp); // # free entries
238 QS_END_PRE()
239
240 tmp = me->tail; // get volatile into temporary
241 QEvt const * const frontEvt = me->ring[tmp];
242
243 Q_ASSERT_INCRIT(420, frontEvt != (QEvt *)0);
244
245 me->frontEvt = frontEvt; // update the original
246
247 if (tmp == 0U) { // need to wrap the tail?
248 tmp = me->end;
249 }
250 --tmp; // advance the tail (counter-clockwise)
251 me->tail = tmp; // update the original
252 }
253 else {
254 me->frontEvt = (QEvt *)0; // queue becomes empty
255
256 // all entries in the queue must be free (+1 for frontEvt)
257 Q_ASSERT_INCRIT(440, tmp == (me->end + 1U));
258
259 QS_BEGIN_PRE(QS_QF_EQUEUE_GET_LAST, qsId)
260 QS_TIME_PRE(); // timestamp
261 QS_SIG_PRE(e->sig); // the signal of this event
262 QS_OBJ_PRE(me); // this queue object
263 QS_2U8_PRE(e->poolNum_, e->refCtr_);
264 QS_END_PRE()
265 }
266 }
267
268 QF_CRIT_EXIT();
269
270 return e;
271}

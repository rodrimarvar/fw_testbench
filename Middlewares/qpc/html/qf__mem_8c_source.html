44//! @public @memberof QMPool
45void QMPool_init(QMPool * const me,
46 void * const poolSto,
47 uint_fast32_t const poolSize,
48 uint_fast16_t const blockSize)
49{
50 QF_CRIT_STAT
51 QF_CRIT_ENTRY();
52
53 Q_REQUIRE_INCRIT(100, poolSto != (void *)0);
54
55 me->freeHead = (void * *)poolSto;
56
57 // find # free links in a memory block, see NOTE1
58 me->blockSize = (QMPoolSize)(2U * sizeof(void *));
59 uint_fast16_t inext = 2U; // # index of the next block
60 while (me->blockSize < (QMPoolSize)blockSize) {
61 me->blockSize += (QMPoolSize)sizeof(void *);
62 ++inext;
63 }
64
65 // the pool buffer must fit at least one rounded-up block
66 Q_ASSERT_INCRIT(110, poolSize >= me->blockSize);
67
68 // start at the head of the free list
69 void * *pfb = me->freeHead; // pointer to free block
70 uint32_t nTot = 1U; // the last block already in the list
71
72 // chain all blocks together in a free-list...
73 for (uint_fast32_t size = poolSize - me->blockSize;
74 size >= (uint_fast32_t)me->blockSize;
75 size -= (uint_fast32_t)me->blockSize)
76 {
77 pfb[0] = &pfb[inext]; // set the next link to next free block
78 pfb = pfb[0]; // advance to the next block
79 ++nTot; // one more free block in the pool
80 }
81 pfb[0] = (void * *)0; // the last link points to NULL
82
83 // dynamic range check
84#if (QF_MPOOL_CTR_SIZE == 1U)
85 Q_ASSERT_INCRIT(190, nTot < 0xFFU);
86#elif (QF_MPOOL_CTR_SIZE == 2U)
87 Q_ASSERT_INCRIT(190, nTot < 0xFFFFU);
88#endif
89
90 me->nTot = (QMPoolCtr)nTot;
91 me->nFree = me->nTot; // all blocks are free
92 me->start = (void * *)poolSto; // the original start this pool buffer
93 me->end = pfb; // the last block in this pool
94 me->nMin = me->nTot; // the minimum # free blocks
95
96 QF_CRIT_EXIT();
97}

100//! @public @memberof QMPool
101void * QMPool_get(QMPool * const me,
102 uint_fast16_t const margin,
103 uint_fast8_t const qsId)
104{
105#ifndef Q_SPY
106 Q_UNUSED_PAR(qsId);
107#endif
108
109 QF_CRIT_STAT
110 QF_CRIT_ENTRY();
111
112 // get volatile into temporaries
113 void * *pfb = me->freeHead; // pointer to free block
114 QMPoolCtr nFree = me->nFree;
115
116 // have more free blocks than the requested margin?
117 if (nFree > (QMPoolCtr)margin) {
118 Q_ASSERT_INCRIT(310, pfb != (void * *)0);
119
120 void * * const pfb_next = pfb[0]; // fast temporary
121
122 --nFree; // one less free block
123 if (nFree == 0U) { // is the pool becoming empty?
124 // pool is becoming empty, so the next free block must be NULL
125 Q_ASSERT_INCRIT(320, pfb_next == (void * *)0);
126
127 me->nFree = 0U; // no more free blocks
128 me->nMin = 0U; // remember that the pool got empty
129 }
130 else { // the pool is NOT empty
131
132 // the next free-block pointer must be in range
133 Q_ASSERT_INCRIT(330,
134 (me->start <= pfb_next) && (pfb_next <= me->end));
135
136 me->nFree = nFree; // update the original
137 if (me->nMin > nFree) { // is this the new minimum?
138 me->nMin = nFree; // remember the minimum so far
139 }
140 }
141
142 me->freeHead = pfb_next; // set the head to the next free block
143
144 // change the allocated block contents so that it is different
145 // than a free block inside the pool.
146 pfb[0] = &me->end[1]; // invalid location beyond the end
147
148 QS_BEGIN_PRE(QS_QF_MPOOL_GET, qsId)
149 QS_TIME_PRE(); // timestamp
150 QS_OBJ_PRE(me); // this memory pool
151 QS_MPC_PRE(nFree); // # free blocks in the pool
152 QS_MPC_PRE(me->nMin); // min # free blocks ever in the pool
153 QS_END_PRE()
154 }
155 else { // don't have enough free blocks at this point
156 pfb = (void * *)0;
157
158 QS_BEGIN_PRE(QS_QF_MPOOL_GET_ATTEMPT, qsId)
159 QS_TIME_PRE(); // timestamp
160 QS_OBJ_PRE(me); // this memory pool
161 QS_MPC_PRE(nFree); // # free blocks in the pool
162 QS_MPC_PRE(margin); // the requested margin
163 QS_END_PRE()
164 }
165
166 QF_CRIT_EXIT();
167
168 return (void *)pfb; // return the block or NULL pointer to the caller
169}

172//! @public @memberof QMPool
173void QMPool_put(QMPool * const me,
174 void * const block,
175 uint_fast8_t const qsId)
176{
177#ifndef Q_SPY
178 Q_UNUSED_PAR(qsId);
179#endif
180
181 void * * const pfb = (void * *)block; // pointer to free block
182
183 QF_CRIT_STAT
184 QF_CRIT_ENTRY();
185
186 // get volatile into temporaries
187 void * * const freeHead = me->freeHead;
188 QMPoolCtr nFree = me->nFree;
189
190 Q_REQUIRE_INCRIT(400, nFree < me->nTot);
191 Q_REQUIRE_INCRIT(410, (me->start <= pfb) && (pfb <= me->end));
192
193 ++nFree; // one more free block in this pool
194
195 me->freeHead = pfb; // set as new head of the free list
196 me->nFree = nFree;
197 pfb[0] = freeHead; // link into the list
198
199 QS_BEGIN_PRE(QS_QF_MPOOL_PUT, qsId)
200 QS_TIME_PRE(); // timestamp
201 QS_OBJ_PRE(me); // this memory pool
202 QS_MPC_PRE(nFree); // the # free blocks in the pool
203 QS_END_PRE()
204
205 QF_CRIT_EXIT();
206}

209// NOTE1:
210// The memory buffer for the pool is organized as an array of void* pointers
211// (see void * data type). These pointers are used to form a linked-list
212// of free blocks in the pool. The first location pfb[0] is the actual link.
213// The second location pfb[1] is used in SafeQP as the redundant "duplicate
214// storage" for the link at pfb[0]. Even though the "duplicate storage" is NOT
215// used in this QP edition, the minimum number of number of void* pointers
216// (void * data type) inside a memory block is still kept at 2 to maintain
217// the same policy for sizing the memory blocks.

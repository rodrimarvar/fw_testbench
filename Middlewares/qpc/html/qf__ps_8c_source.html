62//! @static @private @memberof QActive
63void QActive_publish_(
64 QEvt const * const e,
65 void const * const sender,
66 uint_fast8_t const qsId)
67{
68#ifndef Q_SPY
69 Q_UNUSED_PAR(sender);
70 Q_UNUSED_PAR(qsId);
71#endif
72
73 QSignal const sig = e->sig;
74
75 QF_CRIT_STAT
76 QF_CRIT_ENTRY();
77
78 Q_REQUIRE_INCRIT(200, sig < (QSignal)QActive_maxPubSignal_);
79
80 QS_BEGIN_PRE(QS_QF_PUBLISH, qsId)
81 QS_TIME_PRE(); // the timestamp
82 QS_OBJ_PRE(sender); // the sender object
83 QS_SIG_PRE(sig); // the signal of the event
84 QS_2U8_PRE(e->poolNum_, e->refCtr_);
85 QS_END_PRE()
86
87 // is it a mutable event?
88 if (e->poolNum_ != 0U) {
89 // NOTE: The reference counter of a mutable event is incremented to
90 // prevent premature recycling of the event while the multicasting
91 // is still in progress. At the end of the function, the garbage
92 // collector step (QF_gc()) decrements the reference counter and
93 // recycles the event if the counter drops to zero. This covers the
94 // case when the event was published without any subscribers.
95 Q_ASSERT_INCRIT(205, e->refCtr_ < (2U * QF_MAX_ACTIVE));
96 QEvt_refCtr_inc_(e);
97 }
98
99 // make a local, modifiable copy of the subscriber set
100 QPSet subscrSet = QActive_subscrList_[sig].set;
101
102 QF_CRIT_EXIT();
103
104 if (QPSet_notEmpty(&subscrSet)) { // any subscribers?
105 // highest-prio subscriber
106 uint_fast8_t p = QPSet_findMax(&subscrSet);
107
108 QF_CRIT_ENTRY();
109
110 QActive *a = QActive_registry_[p];
111 // the AO must be registered with the framework
112 Q_ASSERT_INCRIT(210, a != (QActive *)0);
113
114 QF_CRIT_EXIT();
115
116 QF_SCHED_STAT_
117 QF_SCHED_LOCK_(p); // lock the scheduler up to AO's prio
118 do { // loop over all subscribers
119 // QACTIVE_POST() asserts internally if the queue overflows
120 QACTIVE_POST(a, e, sender);
121
122 QPSet_remove(&subscrSet, p); // remove the handled subscriber
123 if (QPSet_notEmpty(&subscrSet)) { // still more subscribers?
124 p = QPSet_findMax(&subscrSet); // highest-prio subscriber
125
126 QF_CRIT_ENTRY();
127
128 a = QActive_registry_[p];
129 // the AO must be registered with the framework
130 Q_ASSERT_INCRIT(220, a != (QActive *)0);
131
132 QF_CRIT_EXIT();
133 }
134 else {
135 p = 0U; // no more subscribers
136 }
137 } while (p != 0U);
138
139 QF_CRIT_ENTRY();
140 Q_ASSERT_INCRIT(290, p == 0U); // all subscribers processed
141 QF_CRIT_EXIT();
142
143 QF_SCHED_UNLOCK_(); // unlock the scheduler
144 }
145
146#if (QF_MAX_EPOOL > 0U)
147 // The following garbage collection step decrements the reference counter
148 // and recycles the event if the counter drops to zero. This covers both
149 // cases when the event was published with or without any subscribers.
150 QF_gc(e); // recycle the event to avoid a leak
151#endif
152}

155//! @protected @memberof QActive
156void QActive_subscribe(QActive const * const me,
157 enum_t const sig)
158{
159 uint_fast8_t const p = (uint_fast8_t)me->prio;
160
161 QF_CRIT_STAT
162 QF_CRIT_ENTRY();
163
164 Q_REQUIRE_INCRIT(300, (Q_USER_SIG <= sig)
165 && (sig < QActive_maxPubSignal_)
166 && (0U < p) && (p <= QF_MAX_ACTIVE)
167 && (QActive_registry_[p] == me));
168
169 QS_BEGIN_PRE(QS_QF_ACTIVE_SUBSCRIBE, p)
170 QS_TIME_PRE(); // timestamp
171 QS_SIG_PRE(sig); // the signal of this event
172 QS_OBJ_PRE(me); // this active object
173 QS_END_PRE()
174
175 // insert the prio. into the subscriber set
176 QPSet_insert(&QActive_subscrList_[sig].set, p);
177
178 QF_CRIT_EXIT();
179}

205 QF_CRIT_EXIT();

209//! @protected @memberof QActive
210void QActive_unsubscribeAll(QActive const * const me) {
211 QF_CRIT_STAT
212 QF_CRIT_ENTRY();
213
214 uint_fast8_t const p = (uint_fast8_t)me->prio;
215
216 Q_REQUIRE_INCRIT(500, (0U < p) && (p <= QF_MAX_ACTIVE)
217 && (QActive_registry_[p] == me));
218 enum_t const maxPubSig = QActive_maxPubSignal_;
219
220 QF_CRIT_EXIT();
221
222 for (enum_t sig = Q_USER_SIG; sig < maxPubSig; ++sig) {
223 QF_CRIT_ENTRY();
224
225 if (QPSet_hasElement(&QActive_subscrList_[sig].set, p)) {
226 QPSet_remove(&QActive_subscrList_[sig].set, p);
227
228 QS_BEGIN_PRE(QS_QF_ACTIVE_UNSUBSCRIBE, p)
229 QS_TIME_PRE(); // timestamp
230 QS_SIG_PRE(sig); // the signal of this event
231 QS_OBJ_PRE(me); // this active object
232 QS_END_PRE()
233 }
234 QF_CRIT_EXIT();
235
236 QF_CRIT_EXIT_NOP(); // prevent merging critical sections
237 }

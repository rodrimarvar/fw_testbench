44//! @protected @memberof QMActive
45void QMActive_ctor(QMActive * const me,
46 QStateHandler const initial)
47{
48 // clear the whole QMActive object, so that the framework can start
49 // correctly even if the startup code fails to clear the uninitialized
50 // data (as is required by the C Standard).
51 QF_bzero_(me, sizeof(*me));
52
53 // NOTE: QActive inherits the QActvie class, but it calls the
54 // constructor of the QMsm subclass. This is because QMActive inherits
55 // the behavior from the QMsm subclass.
56 QMsm_ctor((QMsm *)(me), initial);
57
58 // NOTE: this vtable is identical as QMsm, but is provided
59 // for the QMActive subclass to provide a UNIQUE vptr to distinguish
60 // subclasses of QActive (e.g., in the debugger).
61 static struct QAsmVtable const vtable = { // QMActive virtual table
62 &QMsm_init_,
63 &QMsm_dispatch_,
64 &QMsm_isIn_
65#ifdef Q_SPY
66 ,&QMsm_getStateHandler_
67#endif
68 };
69 me->super.super.vptr = &vtable; // hook vptr to QMActive vtable
70}

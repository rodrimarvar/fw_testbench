52//! @static @public @memberof QK
53QSchedStatus QK_schedLock(uint_fast8_t const ceiling) {
54 QF_CRIT_STAT
55 QF_CRIT_ENTRY();
56
57 Q_REQUIRE_INCRIT(100, !QK_ISR_CONTEXT_());
58
59 // first store the previous lock prio
60 QSchedStatus stat = 0xFFU; // assume scheduler NOT locked
61 if (ceiling > QK_priv_.lockCeil) { // raising the lock ceiling?
62 QS_BEGIN_PRE(QS_SCHED_LOCK, QK_priv_.actPrio)
63 QS_TIME_PRE(); // timestamp
64 // the previous lock ceiling & new lock ceiling
65 QS_2U8_PRE((uint8_t)QK_priv_.lockCeil, (uint8_t)ceiling);
66 QS_END_PRE()
67
68 // previous status of the lock
69 stat = (QSchedStatus)QK_priv_.lockCeil;
70
71 // new status of the lock
72 QK_priv_.lockCeil = ceiling;
73 }
74 QF_CRIT_EXIT();
75
76 return stat; // return the status to be saved in a stack variable
77}

80//! @static @public @memberof QK
81void QK_schedUnlock(QSchedStatus const prevCeil) {
82 // has the scheduler been actually locked by the last QK_schedLock()?
83 if (prevCeil != 0xFFU) {
84 QF_CRIT_STAT
85 QF_CRIT_ENTRY();
86
87 Q_REQUIRE_INCRIT(200, (!QK_ISR_CONTEXT_())
88 && (QK_priv_.lockCeil > prevCeil));
89 QS_BEGIN_PRE(QS_SCHED_UNLOCK, QK_priv_.actPrio)
90 QS_TIME_PRE(); // timestamp
91 // current lock ceiling (old), previous lock ceiling (new)
92 QS_2U8_PRE((uint8_t)QK_priv_.lockCeil, (uint8_t)prevCeil);
93 QS_END_PRE()
94
95 // restore the previous lock ceiling
96 QK_priv_.lockCeil = prevCeil;
97
98 // find if any AOs should be run after unlocking the scheduler
99 if (QK_sched_() != 0U) { // preemption needed?
100 QK_activate_(); // activate any unlocked AOs
101 }
102
103 QF_CRIT_EXIT();
104 }
105}

108//! @static @private @memberof QK
109uint_fast8_t QK_sched_(void) {
110 // NOTE: this function is entered with interrupts DISABLED
111
112 uint_fast8_t p = 0U; // assume NO activation needed
113 if (QPSet_notEmpty(&QK_priv_.readySet)) {
114 // find the highest-prio AO with non-empty event queue
115 p = QPSet_findMax(&QK_priv_.readySet);
116
117 // is the AO's prio. below the active preemption-threshold?
118 if (p <= QK_priv_.actThre) {
119 p = 0U; // no activation needed
120 }
121 else {
122 // is the AO's prio. below the lock-ceiling?
123 if (p <= QK_priv_.lockCeil) {
124 p = 0U; // no activation needed
125 }
126 else {
127 QK_priv_.nextPrio = p; // next AO to run
128 }
129 }
130 }
131
132 return p;
133}

136//! @static @private @memberof QK
137uint_fast8_t QK_sched_act_(
138 QActive const * const act,
139 uint_fast8_t const pthre_in)
140{
141 // NOTE: this function is entered with interrupts DISABLED
142
143 uint_fast8_t p = act->prio;
144 if (act->eQueue.frontEvt == (QEvt *)0) { // empty queue?
145 QPSet_remove(&QK_priv_.readySet, p);
146 }
147
148 if (QPSet_isEmpty(&QK_priv_.readySet)) {
149 p = 0U; // no activation needed
150 }
151 else {
152 // find new highest-prio AO ready to run...
153 p = QPSet_findMax(&QK_priv_.readySet);
154 // NOTE: p is guaranteed to be <= QF_MAX_ACTIVE
155
156 // is the new prio. below the initial preemption-threshold?
157 if (p <= pthre_in) {
158 p = 0U; // no activation needed
159 }
160 else {
161 // is the AO's prio. below the lock preemption-threshold?
162 if (p <= QK_priv_.lockCeil) {
163 p = 0U; // no activation needed
164 }
165 }
166 }
167
168 return p;
169}

172//! @static @private @memberof QK
173void QK_activate_(void) {
174 // NOTE: this function is entered with interrupts DISABLED
175
176 uint_fast8_t const prio_in = QK_priv_.actPrio; // save initial prio.
177 uint_fast8_t p = QK_priv_.nextPrio; // next prio to run
178
179 Q_REQUIRE_INCRIT(500, (prio_in <= QF_MAX_ACTIVE)
180 && (0U < p) && (p <= QF_MAX_ACTIVE));
181
182#if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
183 uint_fast8_t pprev = prio_in;
184#endif // QF_ON_CONTEXT_SW || Q_SPY
185
186 QK_priv_.nextPrio = 0U; // clear for the next time
187
188 uint_fast8_t pthre_in = 0U; // assume preempting the idle thread
189 if (prio_in != 0U) { // preempting NOT the idle thread
190 QActive const * const a = QActive_registry_[prio_in];
191 Q_ASSERT_INCRIT(510, a != (QActive *)0);
192
193 pthre_in = (uint_fast8_t)a->pthre;
194 }
195
196 // loop until no more ready-to-run AOs of higher pthre than the initial
197 do {
198 QActive * const a = QActive_registry_[p];
199 Q_ASSERT_INCRIT(520, a != (QActive *)0); // the AO must be registered
200 uint_fast8_t const pthre = (uint_fast8_t)a->pthre;
201
202 // set new active prio. and preemption-threshold
203 QK_priv_.actPrio = p;
204 QK_priv_.actThre = pthre;
205
206#if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
207 if (p != pprev) { // changing threads?
208
209 QS_BEGIN_PRE(QS_SCHED_NEXT, p)
210 QS_TIME_PRE(); // timestamp
211 QS_2U8_PRE(p, // prio. of the scheduled AO
212 pprev); // previous prio.
213 QS_END_PRE()
214
215#ifdef QF_ON_CONTEXT_SW
216 QF_onContextSw(QActive_registry_[pprev], a);
217#endif // QF_ON_CONTEXT_SW
218
219 pprev = p; // update previous prio.
220 }
221#endif // QF_ON_CONTEXT_SW || Q_SPY
222
223 QF_INT_ENABLE(); // unconditionally enable interrupts
224
225 QEvt const * const e = QActive_get_(a);
226
227 // dispatch event (virtual call)
228 (*a->super.vptr->dispatch)(&a->super, e, p);
229#if (QF_MAX_EPOOL > 0U)
230 QF_gc(e);
231#endif
232
233 // determine the next highest-prio. AO ready to run...
234 QF_INT_DISABLE(); // unconditionally disable interrupts
235 p = QK_sched_act_(a, pthre_in); // schedule next AO
236
237 } while (p != 0U);
238
239 // restore the active prio. and preemption-threshold
240 QK_priv_.actPrio = prio_in;
241 QK_priv_.actThre = pthre_in;
242
243#if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
244 if (prio_in != 0U) { // resuming an active object?
245 QS_BEGIN_PRE(QS_SCHED_NEXT, prio_in)
246 QS_TIME_PRE(); // timestamp
247 // prio. of the resumed AO, previous prio.
248 QS_2U8_PRE(prio_in, pprev);
249 QS_END_PRE()
250
251#ifdef QF_ON_CONTEXT_SW
252 QF_onContextSw(QActive_registry_[pprev],
253 QActive_registry_[prio_in]);
254#endif // QF_ON_CONTEXT_SW
255 }
256 else { // resuming prio.==0 --> idle
257 QS_BEGIN_PRE(QS_SCHED_IDLE, pprev)
258 QS_TIME_PRE(); // timestamp
259 QS_U8_PRE(pprev); // previous prio.
260 QS_END_PRE()
261
262#ifdef QF_ON_CONTEXT_SW
263 QF_onContextSw(QActive_registry_[pprev], (QActive *)0);
264#endif // QF_ON_CONTEXT_SW
265 }
266#endif // QF_ON_CONTEXT_SW || Q_SPY
267}

294//! @static @public @memberof QF
295int_t QF_run(void) {
296#ifdef Q_SPY
297 // produce the QS_QF_RUN trace record
298 QF_INT_DISABLE();
299 QS_beginRec_((uint_fast8_t)QS_QF_RUN);
300 QS_endRec_();
301 QF_INT_ENABLE();
302#endif // Q_SPY
303
304 QF_onStartup(); // application-specific startup callback
305
306 QF_INT_DISABLE();
307
308#ifdef QK_START
309 QK_START(); // port-specific startup of the QK kernel
310#endif
311
312 QK_priv_.lockCeil = 0U; // unlock the QK scheduler
313
314#ifdef QF_ON_CONTEXT_SW
315 // officially switch to the idle context
316 QF_onContextSw((QActive *)0, QActive_registry_[QK_priv_.nextPrio]);
317#endif
318
319 // activate AOs to process events posted so far
320 if (QK_sched_() != 0U) {
321 QK_activate_();
322 }
323
324 QF_INT_ENABLE();
325
326 for (;;) { // QK idle loop...
327 QK_onIdle(); // application-specific QK on-idle callback
328 }
329
330#ifdef __GNUC__
331 return 0;
332#endif
333}

336//! @public @memberof QActive
337void QActive_start(QActive * const me,
338 QPrioSpec const prioSpec,
339 QEvtPtr * const qSto,
340 uint_fast16_t const qLen,
341 void * const stkSto,
342 uint_fast16_t const stkSize,
343 void const * const par)
344{
345 Q_UNUSED_PAR(stkSto); // not needed in QK
346 Q_UNUSED_PAR(stkSize); // not needed in QK
347
348 QF_CRIT_STAT
349 QF_CRIT_ENTRY();
350
351 Q_REQUIRE_INCRIT(300, (me->super.vptr != (struct QAsmVtable *)0)
352 && (stkSto == (void *)0));
353 QF_CRIT_EXIT();
354
355 me->prio = (uint8_t)(prioSpec & 0xFFU); // QF-prio. of the AO
356 me->pthre = (uint8_t)(prioSpec >> 8U); // preemption-threshold
357 QActive_register_(me); // make QF aware of this active object
358
359 QEQueue_init(&me->eQueue, qSto, qLen); // init the built-in queue
360
361 // top-most initial tran. (virtual call)
362 (*me->super.vptr->init)(&me->super, par, me->prio);
363 QS_FLUSH(); // flush the trace buffer to the host
364
365 // See if this AO needs to be scheduled if QK is already running
366 QF_CRIT_ENTRY();
367 if (QK_sched_() != 0U) { // activation needed?
368 QK_activate_();
369 }
370 QF_CRIT_EXIT();
371}

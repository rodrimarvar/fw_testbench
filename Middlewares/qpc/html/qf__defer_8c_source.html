44//! @protected @memberof QActive
45bool QActive_defer(QActive const * const me,
46 struct QEQueue * const eq,
47 QEvt const * const e)
48{
49 bool const status = QEQueue_post(eq, e, 0U, me->prio);
50
51 QS_CRIT_STAT
52 QS_CRIT_ENTRY();
53 QS_BEGIN_PRE(QS_QF_ACTIVE_DEFER, me->prio)
54 QS_TIME_PRE(); // time stamp
55 QS_OBJ_PRE(me); // this active object
56 QS_OBJ_PRE(eq); // the deferred queue
57 QS_SIG_PRE(e->sig); // the signal of the event
58 QS_2U8_PRE(e->poolNum_, e->refCtr_);
59 QS_END_PRE()
60 QS_CRIT_EXIT();
61
62 return status;
63}

66//! @protected @memberof QActive
67bool QActive_recall(QActive * const me,
68 struct QEQueue * const eq)
69{
70 QEvt const * const e = QEQueue_get(eq, me->prio);
71 QF_CRIT_STAT
72
73 bool recalled = false;
74 if (e != (QEvt *)0) { // event available?
75 QACTIVE_POST_LIFO(me, e); // post it to the front of the AO's queue
76
77 QF_CRIT_ENTRY();
78
79 if (e->poolNum_ != 0U) { // is it a mutable event?
80
81 // after posting to the AO's queue the event must be referenced
82 // at least twice: once in the deferred event queue (eq->get()
83 // did NOT decrement the reference counter) and once in the
84 // AO's event queue.
85 Q_ASSERT_INCRIT(205, e->refCtr_ >= 2U);
86
87 // we need to decrement the reference counter once, to account
88 // for removing the event from the deferred event queue.
89 QEvt_refCtr_dec_(e); // decrement the reference counter
90 }
91
92 QS_BEGIN_PRE(QS_QF_ACTIVE_RECALL, me->prio)
93 QS_TIME_PRE(); // time stamp
94 QS_OBJ_PRE(me); // this active object
95 QS_OBJ_PRE(eq); // the deferred queue
96 QS_SIG_PRE(e->sig); // the signal of the event
97 QS_2U8_PRE(e->poolNum_, e->refCtr_);
98 QS_END_PRE()
99
100 QF_CRIT_EXIT();
101
102 recalled = true;
103 }
104 else {
105 QS_CRIT_ENTRY();
106
107 QS_BEGIN_PRE(QS_QF_ACTIVE_RECALL_ATTEMPT, me->prio)
108 QS_TIME_PRE(); // time stamp
109 QS_OBJ_PRE(me); // this active object
110 QS_OBJ_PRE(eq); // the deferred queue
111 QS_END_PRE()
112
113 QS_CRIT_EXIT();
114 }
115 return recalled;
116}

119//! @protected @memberof QActive
120uint_fast16_t QActive_flushDeferred(QActive const * const me,
121 struct QEQueue * const eq,
122 uint_fast16_t const num)
123{
124 uint_fast16_t n = 0U;
125 while (n < num) {
126 QEvt const * const e = QEQueue_get(eq, me->prio);
127 if (e != (QEvt *)0) {
128 ++n; // count one more flushed event
129#if (QF_MAX_EPOOL > 0U)
130 QF_gc(e); // garbage collect
131#endif
132 }
133 else {
134 break;
135 }
136 }
137
138 return n;
139}
